## 常用的质数筛法(lc版)

### 埃式筛:

```cpp
const int MX = 1'000'001;
bool is_prime[MX];
vector<int> primes;
auto init = [] {
    ranges::fill(is_prime, true);
    is_prime[0] = is_prime[1] = false;
    for (int i = 2; i < MX; i++) {
        if (is_prime[i]) {
            primes.push_back(i);
            for (long long j = 1LL * i * i; j < MX; j += i) {
                is_prime[j] = false;
            }
        }
    }
    return 0;
}();
```

### 欧拉线性筛:

```cpp
const int MX = 1'000'001;
bool is_prime[MX];
vector<int> primes;
auto init = [] {
    ranges::fill(is_prime, true);
    is_prime[0] = is_prime[1] = false;
    for (int i = 2; i < MX; i++) {
        if (is_prime[i]) primes.push_back(i);
        for (int j = 0; j < primes.size() && i*primes[j]< MX; j++) {
            is_prime[i*primes[j]] = false;
            if (i % primes[j] ==0 ) break;
        }
    }
    return 0;
}();
```

## 判断是否为平方数

```cpp
bool is_square(ll x) {
    if (x < 0) return false;
    int d = x % 10;
    if(d != 0 && d != 1 && d != 4 && d != 5 && d != 6 && d != 9) return false;
    if (x % 4 == 2 || x % 4 == 3) return false;
    ll s = sqrt(x);
    while ((s + 1) * (s + 1) <= x) s++;
    while (s * s > x) s--;
    return s * s == x;
}
```
